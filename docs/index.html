影片中逐步示範的經典程式碼

# 1. 匯入 PyGMT
import pygmt

# 2. 建立第一張基本地圖（全球 Mercator）
fig = pygmt.Figure()
fig.basemap(
    region=[-180, 180, -70, 70],
    projection="M20c",
    frame=True
)
fig.show()
# 3. 加海岸線 + 陸地/海洋上色（Hammer 投影）
fig = pygmt.Figure()
fig.basemap(region="d", projection="H", frame=True)
fig.coast(
    shorelines=True,
    land="darkgreen",
    water="lightsteelblue"
)
fig.show()
# 4. 顯示全球地形（使用遠端 SRTM15+ 資料集）
fig = pygmt.Figure()
fig.basemap(region="d", projection="H", frame=True)
fig.grdimage(
    grid="@earth_relief_20m",
    shading=True
)
fig.show()
